function renderList() { const area = document.getElementById('renderArea'); if(!area) return; area.innerHTML = ''; Object.values(timers).forEach(clearInterval); timers = {}; const all = Object.entries(window.currentData); const auth = hasAuth(); const drawBox = (items, label) => { if (items.length === 0) return; const sec = document.createElement('div'); sec.innerHTML = `
${label}
`; area.appendChild(sec); const grid = sec.querySelector('.grid'); items.forEach(([id, b]) => { const st = getBossStatus(b, id); const card = document.createElement('div'); card.className = 'card'; // 修正內容：將按鈕文字與顯示邏輯精確化 const btnText = b.displayValue === 'ON' ? 'KILL' : 'NEXT'; card.innerHTML = `${auth ? `` : ''}

${b.map} - ${b.ch}

...
${(b.targetTime === 0 && st.weight !== 7) ? `` : ''}
Last: ${b.updater || ''}
`; grid.appendChild(card); startTimer(id, b, st); // 修正：傳入整個 st 物件 }); }; const sortedAll = sortItems(all); const pinned = sortedAll.filter(i => getBossStatus(i[1], i[0]).pinned); const others = sortedAll.filter(i => !getBossStatus(i[1], i[0]).pinned); drawBox(pinned, "置頂追蹤區 ★"); const killed = others.filter(i => getBossStatus(i[1], i[0]).weight === 7); const active = others.filter(i => getBossStatus(i[1], i[0]).weight !== 7); drawBox(killed, "已擊殺區間"); if (currentSort === 'time') { drawBox(active, "全域時間排序 (閒置置底)"); } else if (currentSort === 'map') { drawBox(active.sort((a,b) => getBossStatus(b[1], b[0]).mapLv - getBossStatus(a[1], a[0]).mapLv || a[1].ch - b[1].ch), "全域地圖等級排序"); } else { const stages = [], timersList = [], idleList = []; active.forEach(([id, b]) => { const st = getBossStatus(b, id); if (st.isGrey) idleList.push([id, b]); else if (b.targetTime > 0) timersList.push([id, b]); else stages.push([id, b]); }); drawBox(stages, "階段追蹤區"); drawBox(timersList, "重生計時區"); drawBox(idleList, "超時閒置區"); } } function startTimer(id, b, st) { // 修正：參數接收 st 物件 const up = () => { const el = document.getElementById(`t_${id}`); if(!el) return; const now = Date.now(); if (b.targetTime > 0) { const d = Math.floor((b.targetTime - now) / 1000), a = Math.abs(d); el.innerText = `${Math.floor(a/3600)}:${Math.floor((a%3600)/60).toString().padStart(2,'0')}:${(a%60).toString().padStart(2,'0')}`; if (!st.isGrey) el.style.color = d <= 0 ? '#ff3300' : '#00FF00'; } else { const m = Math.floor((now - b.startTime) / 60000), s = Math.floor(((now - b.startTime) % 60000) / 1000); const mat = b.displayValue.match(/[\d.]+/); const lv = b.displayValue === "ON" ? 99 : (mat ? Math.floor(parseFloat(mat[0])) : 0); const isW = (lv < 2 && m >= 12) || (lv >= 2 && lv < 3 && m >= 7) || (lv >= 3 && lv < 4 && m >= 5) || (lv === 4 && m >= 3) || (lv === 99 && m >= 3); el.innerHTML = `${b.displayValue}(${m}:${s.toString().padStart(2,'0')})${(showWarningIcon && isW && !st.isGrey && st.weight !== 7) ? '⚠️' : ''}`; if (!st.isGrey) el.style.color = st.color; } }; up(); timers[id] = setInterval(up, 1000); }
