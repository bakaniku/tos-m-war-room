// --- 以下為修改後的腳本部分，請替換原 script 標籤內的相關函數 --- // 修正：增加全域音量獲取與語音播報功能 function updateVolBg(el) { const val = (el.value - el.min) / (el.max - el.min) * 100; el.style.backgroundSize = val + '% 100%'; } function unlockAudio() { if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)(); if (audioCtx.state === 'suspended') audioCtx.resume(); } // 修正：使叮咚聲受音量滑桿控制 function playDing(isTest = false) { if (!isSfxEnabled && !isTest) return; // 如果沒開啟且不是測試模式就不響 unlockAudio(); const vol = parseFloat(document.getElementById('volRange').value); const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); osc.type = 'sine'; osc.frequency.setValueAtTime(880, audioCtx.currentTime); osc.connect(gain); gain.connect(audioCtx.destination); gain.gain.setValueAtTime(0, audioCtx.currentTime); // 音量線性衰減，最高 0.3 (避免爆音) 乘以滑桿比例 gain.gain.linearRampToValueAtTime(vol * 0.3, audioCtx.currentTime + 0.01); osc.start(); osc.stop(audioCtx.currentTime + 0.3); } // 修正：語音播報函數 function speak(text, isTest = false) { if (!isVoiceEnabled && !isTest) return; window.speechSynthesis.cancel(); // 停止之前的播報 const vol = parseFloat(document.getElementById('volRange').value); const msg = new SpeechSynthesisUtterance(text); msg.volume = vol; msg.lang = 'zh-TW'; msg.rate = 1.1; // 語速稍微加快 window.speechSynthesis.speak(msg); } // 修正：開關點擊時觸發測試音 function toggleVoice() { isVoiceEnabled = !isVoiceEnabled; document.getElementById('voiceBtn').classList.toggle('active', isVoiceEnabled); unlockAudio(); if (isVoiceEnabled) speak("語音開啟", true); // 測試播報 } function toggleSfx() { isSfxEnabled = !isSfxEnabled; document.getElementById('sfxBtn').classList.toggle('active', isSfxEnabled); unlockAudio(); if (isSfxEnabled) playDing(true); // 測試鈴聲 } // 修改：在 Firebase 監聽處加入語音播報邏輯 db.ref(`rooms/${currentRoom}/bosses`).on('value', s => { const incomingData = s.val() || {}; Object.keys(incomingData).forEach(k => { const boss = incomingData[k]; const inputVal = boss.lastInput.toString().toUpperCase(); if (inputVal !== lastProcessedInput[k] && boss.isFlash) { if (inputVal === "ON") { playDing(); speak(`${boss.map}分流${boss.ch}重生`); } else if (inputVal.startsWith("R")) { const num = parseFloat(inputVal.substring(1)) || 0; if (num >= 4.0) { playDing(); speak(`${boss.map}分流${boss.ch}階段${num}`); } } } lastProcessedInput[k] = inputVal; }); window.currentData = incomingData; renderList(); });
